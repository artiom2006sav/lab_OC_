#!/bin/bash
counter=0
trap 'echo "SIGTERM – завершение"; exit 0' SIGTERM
trap 'echo "SIGUSR1 – сброс счётчика"; counter=0' SIGUSR1
trap 'echo "Ctrl+C нажат, используйте SIGTERM"' SIGINT
echo "PID: $$"
while true; do
    ((counter++))
    echo "Счётчик: $counter"
    sleep 1
done

sudo tee /etc/systemd/system/signal_demo.service >/dev/null <<EOF
[Unit]
Description=Signal Demo
After=network.target

[Service]
Type=simple
User=$USER
WorkingDirectory=$HOME/lab_scripts
ExecStart=$HOME/lab_scripts/signal_demo.sh
Restart=on-failure
RestartSec=5
StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target
EOF

sudo systemctl daemon-reload
sudo systemctl enable --now signal_demo.service
sudo systemctl status signal_demo.service
journalctl -u signal_demo.service -f   # Ctrl+C для выхода

#!/bin/bash
TAG="sysmon"
CPU=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d% -f1)
MEM=$(free | grep Mem | awk '{print $3/$2*100}')
[[ $(echo "$CPU > 80" | bc) -eq 1 ]] && logger -t "$TAG" "WARNING: CPU $CPU%"
[[ $(echo "$MEM > 90" | bc) -eq 1 ]] && logger -t "$TAG" "CRITICAL: MEM $MEM%"
for s in ssh cron; do
    systemctl is-active --quiet $s || logger -t "$TAG" "ALERT: $s not running"
done

cron 
chmod +x ~/lab_scripts/system_monitor.sh
(crontab -l 2>/dev/null; echo "*/5 * * * * $HOME/lab_scripts/system_monitor.sh") | crontab -

sudo tee /etc/rsyslog.d/50-sysmon.conf >/dev/null <<EOF
if \$programname == 'sysmon' then /var/log/sysmon.log
& stop
EOF
sudo systemctl restart rsyslog
