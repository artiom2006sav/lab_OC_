#!/bin/bash
LOG="/var/log/user_audit_$(date +%Y%m%d).log"
exec > >(tee -a "$LOG") 2>&1

echo "=== Отчёт $(date) | Хост $(hostname) | Пользователь $USER ==="

# Статистика
echo -e "\n--- Общая статистика ---"
echo "Всего: $(wc -l < /etc/passwd)"
echo "Системных (UID<1000): $(awk -F: '$3<1000' /etc/passwd | wc -l)"
echo "Обычных (UID>=1000): $(awk -F: '$3>=1000' /etc/passwd | wc -l)"
echo "Заблокированных: $(sudo awk -F: '$2~/^[!*]/' /etc/shadow | wc -l)"
echo "Пустых паролей: $(sudo awk -F: '$2==""' /etc/shadow | wc -l)"

# Угрозы
echo -e "\n--- Потенциальные угрозы ---"
echo "UID=0 (кроме root): $(awk -F: '$3==0 && $1!="root"' /etc/passwd | cut -d: -f1)"
echo "Без пароля: $(sudo awk -F: '$2==""' /etc/shadow | cut -d: -f1)"
echo "Bash/sh без home: $(awk -F: '($7=="/bin/bash"||$7=="/bin/sh") && !system("test -d "$6)' /etc/passwd | cut -d: -f1)"
echo ".rhosts/.netrc: $(find /home -name ".rhosts" -o -name ".netrc" 2>/dev/null)"
echo "Не входили >90 дней: $(sudo awk -F: '{if($3>0 && (systime()/86400 - $3)>90) print $1}' /etc/shadow)"

# Политики
echo -e "\n--- Политики паролей ---"
echo "Истёкший пароль: $(sudo awk -F: '{if($5>0 && $3+$5 < systime()/86400) print $1}' /etc/shadow)"
echo "Никогда не истекает: $(sudo awk -F: '{if($5==99999 || $5==-1) print $1}' /etc/shadow)"
echo "Истекает <7 дней: $(sudo awk -F: '{if($5>0 && $3+$5 < systime()/86400+7) print $1}' /etc/shadow)"

# Группы
echo -e "\n--- Группы ---"
echo "Пустые группы: $(awk -F: '$4==""' /etc/group | cut -d: -f1)"
echo "Дубли GID: $(awk -F: '{print $3}' /etc/group | sort | uniq -d)"
echo "В sudo/wheel: $(grep -E "^(sudo|wheel):" /etc/group | cut -d: -f4 | tr ',' '\n' | sort -u)"

echo -e "\nОтчёт сохранён в $LOG"

#!/bin/bash
counter=0
trap 'echo "SIGTERM – завершение"; exit 0' SIGTERM
trap 'echo "SIGUSR1 – сброс счётчика"; counter=0' SIGUSR1
trap 'echo "Ctrl+C нажат, используйте SIGTERM"' SIGINT
echo "PID: $$"
while true; do
    ((counter++))
    echo "Счётчик: $counter"
    sleep 1
done

#!/bin/bash
TAG="sysmon"
CPU=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d% -f1)
MEM=$(free | grep Mem | awk '{print $3/$2*100}')
[[ $(echo "$CPU > 80" | bc) -eq 1 ]] && logger -t "$TAG" "WARNING: CPU $CPU%"
[[ $(echo "$MEM > 90" | bc) -eq 1 ]] && logger -t "$TAG" "CRITICAL: MEM $MEM%"
for s in ssh cron; do
    systemctl is-active --quiet $s || logger -t "$TAG" "ALERT: $s not running"
done
