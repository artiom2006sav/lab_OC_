cat > /tmp/signal_test.sh << 'EOF'
#!/bin/bash
counter=0
trap 'echo "SIGTERM – завершение..."; exit 0' SIGTERM
trap 'echo "SIGUSR1 – время: $(date), счётчик: $counter"' SIGUSR1
trap 'echo "SIGUSR2 – сброс счётчика"; counter=0' SIGUSR2
trap 'echo "Ctrl+C игнорируется"' SIGINT
echo "PID: $$"
while true; do
    ((counter++))
    sleep 1
done
EOF
chmod +x /tmp/signal_test.sh
/tmp/signal_test.sh &
PID=$!
kill -USR1 $PID
kill -USR1 $PID
kill -USR2 $PID
kill -USR1 $PID
kill -TERM $PID

4.2 
sudo useradd -r -s /usr/sbin/nologin webapp
sudo mkdir -p /opt/webapp/logs
sudo chown webapp:webapp /opt/webapp /opt/webapp/logs

sudo tee /opt/webapp/app.sh > /dev/null << 'EOF'
#!/bin/bash
LOG="/opt/webapp/logs/app.log"
counter=0
trap 'echo "$(date) Получен SIGTERM" >> $LOG; exit 0' SIGTERM
trap 'echo "$(date) Перезагрузка конфигурации" >> $LOG' SIGHUP
while true; do
    ((counter++))
    echo "$(date) Итерация $counter" >> $LOG
    sleep 5
done
EOF
sudo chmod +x /opt/webapp/app.sh
sudo chown webapp:webapp /opt/webapp/app.sh

sudo tee /etc/systemd/system/webapp.service > /dev/null << 'EOF'
[Unit]
Description=WebApp Service
After=network.target

[Service]
Type=simple
User=webapp
WorkingDirectory=/opt/webapp
ExecStart=/opt/webapp/app.sh
Restart=on-failure
RestartSec=5
StartLimitInterval=60
StartLimitBurst=3
MemoryMax=512M
CPUQuota=50%

[Install]
WantedBy=multi-user.target
EOF

sudo systemctl daemon-reload
sudo systemctl enable --now webapp.service
sudo systemctl status webapp.service
journalctl -u webapp.service -f &
sudo kill -HUP $(pgrep -f /opt/webapp/app.sh)
sudo kill -TERM $(pgrep -f /opt/webapp/app.sh)
sleep 6
sudo systemctl status webapp.service
